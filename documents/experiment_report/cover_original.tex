% cover_original.tex
% 封面组件：在 main.tex 导言区使用 \input{cover_original.tex}
% 正文中使用 \MakeExperimentCover 生成封面。
%
% 本文件不包含 \documentclass、\begin{document}、\end{document}。

\RequirePackage{graphicx}
\RequirePackage{tikz}
\RequirePackage{eso-pic}
\RequirePackage{adjustbox}

% ============================================================
% 用户配置区：只需修改这里
% ============================================================

% 图片文件名
\def\CoverSchoolImage{cover_school_name.png}
\def\CoverTitleImage{cover_title.png}

% 封面信息
\def\CoverDepartment{}
\def\CoverMajor{}
\def\CoverGrade{}
\def\CoverCourse{}
\def\CoverStudentID{}
\def\CoverStudentName{}
\def\CoverInstructor{}

% 日期
\def\CoverYear{}
\def\CoverMonth{}
\def\CoverDay{}

% 填写内容相对下划线的上移距离
% 数值越大，文字离下划线越远
\def\CoverContentLift{4pt}

% 常用尺寸
\def\CoverSchoolImageWidth{306.1pt}
\def\CoverTitleImageWidth{229pt}
\def\CoverLineWidth{202.4pt}
\def\CoverDateWidth{150pt}

% ============================================================
% 高级版式区：一般不需要修改
% ============================================================

\def\CoverMainTextStyle{%
    \fontsize{15pt}{18pt}\selectfont
}

% A4 页面水平中心
\def\CoverPageCenterX{298.75}

% 图片纵向位置
\def\CoverSchoolImageY{169}
\def\CoverTitleImageY{293.5}

% 主信息栏横向位置
\def\CoverLabelX{153}
\def\CoverLineStartX{228}
\def\CoverLineEndX{430.4}

% 主信息栏纵向位置
\def\CoverDepartmentY{422.15}
\def\CoverMajorY{453.35}
\def\CoverGradeY{484.55}
\def\CoverCourseY{515.75}
\def\CoverStudentIDY{546.95}
\def\CoverStudentNameY{578.15}
\def\CoverInstructorY{609.35}

% 日期纵向位置
\def\CoverDateY{687}

% ============================================================
% 内部实现：无需修改
% ============================================================

% #1：字段名称
% #2：字段内容
% #3：距离页面顶部的位置，单位 pt
\newcommand{\CoverOriginalField}[3]{%
    \node[
        anchor=base west,
        inner sep=0pt,
        yshift=1.25pt
    ] at (\CoverLabelX,-#3) {%
        \CoverMainTextStyle #1%
    };

    \draw[
        line width=0.72pt
    ]
        (\CoverLineStartX,-#3)
        --
        (\CoverLineEndX,-#3);

    \node[
        anchor=base west,
        inner sep=0pt,
        yshift=\CoverContentLift
    ] at (\CoverLineStartX,-#3) {%
        \makebox[\CoverLineWidth][c]{%
            \adjustbox{max width=\CoverLineWidth}{%
                \CoverMainTextStyle #2%
            }%
        }%
    };
}

\newcommand{\MakeExperimentCover}{%
    \thispagestyle{empty}

    \AddToShipoutPictureFG*{%
        \AtPageUpperLeft{%
            \begin{tikzpicture}[
                overlay,
                x=1pt,
                y=1pt
            ]

                % 学校名称图片
                \node[
                    anchor=north,
                    inner sep=0pt
                ] at (
                    \CoverPageCenterX,
                    -\CoverSchoolImageY
                ) {%
                    \includegraphics[
                        width=\CoverSchoolImageWidth
                    ]{\CoverSchoolImage}%
                };

                % “学生实验报告”标题图片
                \node[
                    anchor=north,
                    inner sep=0pt
                ] at (
                    \CoverPageCenterX,
                    -\CoverTitleImageY
                ) {%
                    \includegraphics[
                        width=\CoverTitleImageWidth
                    ]{\CoverTitleImage}%
                };

                % 主信息栏
                \CoverOriginalField
                    {院\hspace{30pt}系：}
                    {\CoverDepartment}
                    {\CoverDepartmentY}

                \CoverOriginalField
                    {专\hspace{30pt}业：}
                    {\CoverMajor}
                    {\CoverMajorY}

                \CoverOriginalField
                    {年\hspace{30pt}级：}
                    {\CoverGrade}
                    {\CoverGradeY}

                \CoverOriginalField
                    {课程名称：}
                    {\CoverCourse}
                    {\CoverCourseY}

                \CoverOriginalField
                    {学\hspace{30pt}号：}
                    {\CoverStudentID}
                    {\CoverStudentIDY}

                \CoverOriginalField
                    {姓\hspace{30pt}名：}
                    {\CoverStudentName}
                    {\CoverStudentNameY}

                \CoverOriginalField
                    {指导教师：}
                    {\CoverInstructor}
                    {\CoverInstructorY}

                % 日期
                \node[
                    anchor=base,
                    inner sep=0pt
                ] at (
                    \CoverPageCenterX,
                    -\CoverDateY
                ) {%
                    \CoverMainTextStyle
                    \makebox[\CoverDateWidth][s]{%
                        \CoverYear\ 年
                        \hfill
                        \CoverMonth\ 月
                        \hfill
                        \CoverDay\ 日
                    }%
                };

            \end{tikzpicture}%
        }%
    }%

    \null
    \clearpage
}
