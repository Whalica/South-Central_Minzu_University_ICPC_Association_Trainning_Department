% #1：字段名称
% #2：字段内容
% #3：距离页面顶部的位置，单位为 pt
\newcommand{\CoverField}[3]{%
    % 左侧字段名称
    \node[
        anchor=base west,
        inner sep=0pt,
        yshift=1.25pt
    ] at (\CoverLabelX,-#3) {%
        \CoverMainTextStyle #1%
    };

    % 固定长度下划线
    \draw[
        line width=0.72pt
    ]
        (\CoverLineStartX,-#3)
        --
        (\CoverLineEndX,-#3);

    % 固定宽度内容区
    % 填写内容不会改变下划线长度
    \node[
        anchor=base west,
        inner sep=0pt,
        yshift=\CoverContentLift
    ] at (\CoverLineStartX,-#3) {%
        \makebox[\CoverLineWidth][c]{%
            \adjustbox{max width=\CoverLineWidth}{%
                \CoverMainTextStyle #2%
            }%
        }%
    };
}

\newcommand{\MakeExperimentCover}{%
    \thispagestyle{empty}

    \AddToShipoutPictureFG*{%
        \AtPageUpperLeft{%
            \begin{tikzpicture}[
                overlay,
                x=1pt,
                y=1pt
            ]

                % 学校名称图片
                \node[
                    anchor=north,
                    inner sep=0pt
                ] at (
                    \CoverPageCenterX,
                    -\CoverSchoolImageY
                ) {%
                    \includegraphics[
                        width=\CoverSchoolImageWidth
                    ]{\CoverSchoolImage}%
                };

                % “学生实验报告”标题图片
                \node[
                    anchor=north,
                    inner sep=0pt
                ] at (
                    \CoverPageCenterX,
                    -\CoverTitleImageY
                ) {%
                    \includegraphics[
                        width=\CoverTitleImageWidth
                    ]{\CoverTitleImage}%
                };

                % 主信息栏
                \CoverField
                    {院\hspace{30pt}系：}
                    {\CoverDepartment}
                    {\CoverDepartmentY}

                \CoverField
                    {专\hspace{30pt}业：}
                    {\CoverMajor}
                    {\CoverMajorY}

                \CoverField
                    {年\hspace{30pt}级：}
                    {\CoverGrade}
                    {\CoverGradeY}

                \CoverField
                    {课程名称：}
                    {\CoverCourse}
                    {\CoverCourseY}

                \CoverField
                    {学\hspace{30pt}号：}
                    {\CoverStudentID}
                    {\CoverStudentIDY}

                \CoverField
                    {姓\hspace{30pt}名：}
                    {\CoverStudentName}
                    {\CoverStudentNameY}

                \CoverField
                    {指导教师：}
                    {\CoverInstructor}
                    {\CoverInstructorY}

                % 日期
                \node[
                    anchor=base,
                    inner sep=0pt
                ] at (
                    \CoverPageCenterX,
                    -\CoverDateY
                ) {%
                    \CoverMainTextStyle
                    \makebox[\CoverDateWidth][s]{%
                        \CoverYear\ 年
                        \hfill
                        \CoverMonth\ 月
                        \hfill
                        \CoverDay\ 日
                    }%
                };

            \end{tikzpicture}%
        }%
    }%

    \null
    \clearpage
}
